function encrypt(plaintext):
    ciphertext = ""
    for each char P in plaintext:
        if P is an alphabet letter:
            if P == 'J': P = 'I' // Handle I/J merge
            (row, col) = char_to_coords(P)

        else:
            ciphertext += P // Non-alphabet characters remain unchanged
    return ciphertext
