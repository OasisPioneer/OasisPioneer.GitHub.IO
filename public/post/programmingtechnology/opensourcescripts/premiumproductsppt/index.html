def get_total_pages(category_url):
    """‰øÆÂ§çÂàÜÈ°µURLÊ†ºÂºèÔºå‰ΩøÁî®ËøûÂ≠óÁ¨¶ËÄåÈùû‰∏ãÂàíÁ∫ø"""
    print(f"  - Ê≠£Âú®Êé¢ÊµãÂàÜÁ±ª ‚Äú{category_url}‚Äù ÁöÑÊÄªÈ°µÊï∞...")
    total_pages = 1
    base_url = f"{BASE_URL}{category_url}".rstrip('/')
    
    while True:
        # ÂÖ≥ÈîÆ‰øÆÂ§çÔºöÂàÜÈ°µURL‰ΩøÁî®ËøûÂ≠óÁ¨¶ "list-{page}.html" ËÄåÈùû‰∏ãÂàíÁ∫ø
        current_url = base_url if total_pages == 1 else f"{base_url}/list-{total_pages}.html"
        print(f"    Êé¢ÊµãÁ¨¨{total_pages}È°µ: {current_url}")  # Ë∞ÉËØïËæìÂá∫ÂΩìÂâçURL
        try:
            response = requests.get(current_url, headers=HEADERS, timeout=10)
            print(f"    ÂìçÂ∫îÁä∂ÊÄÅ: {response.status_code}")  # ÊòæÁ§∫ÂìçÂ∫îÁä∂ÊÄÅÁ†Å
            
            if response.status_code == 404:
                print(f"    Á¨¨{total_pages}È°µ‰∏çÂ≠òÂú®ÔºåÂÅúÊ≠¢Êé¢Êµã")
                break
            
            response.raise_for_status()
            response.encoding = 'utf-8'
            soup = BeautifulSoup(response.text, 'html.parser')
            
            # Ê†πÊçÆÈ°µÈù¢Ê∫êÁ†Å‰ºòÂåñ‰∏ã‰∏ÄÈ°µÊåâÈíÆËØÜÂà´
            next_page = soup.find('a', string='‰∏ã‰∏ÄÈ°µ')
            
            if not next_page:
                print(f"    Êú™ÊâæÂà∞‰∏ã‰∏ÄÈ°µÊåâÈíÆÔºåÂΩìÂâçÈ°µ{total_pages}‰∏∫ÊúÄÂêé‰∏ÄÈ°µ")
                break
            
            # È™åËØÅ‰∏ã‰∏ÄÈ°µURLÊòØÂê¶ÊúâÊïàÔºàÈò≤Ê≠¢ËôöÂÅáÊåâÈíÆÔºâ
            next_page_url = next_page.get('href', '')
            if not next_page_url or 'list-' not in next_page_url:
                print(f"    ‰∏ã‰∏ÄÈ°µÈìæÊé•Êó†ÊïàÔºåÂÅúÊ≠¢Êé¢Êµã")
                break
            
            total_pages += 1
            time.sleep(random.uniform(1, 1.5))  # Âª∂ËøüÈÅøÂÖçÂèçÁà¨
            
        except Exception as e:
            print(f"    Êé¢ÊµãÁ¨¨{total_pages}È°µÂá∫Èîô: {e}ÔºåÂÅúÊ≠¢Êé¢Êµã")
            break
    
    print(f"  - ‚úì Êé¢ÊµãÂà∞ÊÄªÈ°µÊï∞‰∏∫: {total_pages}")
    return total_pages

def get_ppt_list_from_category(category_url, max_pages):
    """‰øÆÂ§çÂàóË°®È°µURLÊ†ºÂºèÔºåÂåπÈÖçÁΩëÁ´ôÂÆûÈôÖÂàÜÈ°µ"""
    all_detail_urls = []
    for page_num in range(1, max_pages + 1):
        base_cat_url = category_url.rstrip('/')
        # ‰øÆÂ§çÂàóË°®È°µURLÊ†ºÂºèÔºö‰ΩøÁî®ËøûÂ≠óÁ¨¶
        list_url = f"{BASE_URL}{base_cat_url}" if page_num == 1 else f"{BASE_URL}{base_cat_url}/list-{page_num}.html"
        
        print(f"  - Ê≠£Âú®ÂàÜÊûêÁ¨¨ {page_num}/{max_pages} È°µ: {list_url}")
        try:
            response = requests.get(list_url, headers=HEADERS)
            if response.status_code == 404:
                print("  - ÊèêÁ§∫: È°µÈù¢‰∏çÂ≠òÂú®ÔºåÂ∑≤Âà∞ËææÊúÄÂêé‰∏ÄÈ°µ„ÄÇ")
                break
            response.raise_for_status()
            response.encoding = 'utf-8'
            soup = BeautifulSoup(response.text, 'html.parser')
            ppt_links = soup.select('ul.posts.clear > li > a.p-title')
            if not ppt_links:
                print("  - ÊèêÁ§∫: Ê≠§È°µÊú™ÊâæÂà∞PPTÈìæÊé•ÔºåÂ∑≤Âà∞ËææÊúÄÂêé‰∏ÄÈ°µ„ÄÇ")
                break
            all_detail_urls.extend([link['href'] for link in ppt_links])
            time.sleep(random.uniform(0.5, 1.5))
        except Exception as e:
            print(f"  - ÈîôËØØ: ÂàÜÊûêÂàóË°®È°µ {list_url} Êó∂Âá∫Èîô - {e}")
            break
    return all_detail_urls

def get_download_page_url(detail_page_url):
    try:
        full_url = f"{BASE_URL}{detail_page_url}"
        response = requests.get(full_url, headers=HEADERS)
        response.raise_for_status()
        response.encoding = 'utf-8'
        soup = BeautifulSoup(response.text, 'html.parser')
        down_button = soup.select_one('a.down-button')
        return down_button['href'] if down_button and down_button.has_attr('href') else None
    except Exception as e:
        print(f"  - ÈîôËØØ: ËÆøÈóÆËØ¶ÊÉÖÈ°µ {full_url} Êó∂Âá∫Èîô - {e}")
        return None

def download_ppt_file(download_page_url, save_folder):
    try:
        full_url = f"{BASE_URL}{download_page_url}"
        response = requests.get(full_url, headers=HEADERS)
        response.raise_for_status()
        response.encoding = 'utf-8'
        soup = BeautifulSoup(response.text, 'html.parser')
        title = (soup.select_one('div.de > h1') or soup.new_tag('h1')).text.strip().replace(' - ‰∏ãËΩΩÈ°µ', '') or f"ppt_{random.randint(1000, 9999)}"
        final_link_tag = soup.select_one('ul.down.clear > li > a')
        if not final_link_tag or not final_link_tag.has_attr('href'):
            print(f"  - Ë≠¶ÂëäÔºöÂú®‰∏ãËΩΩÈ°µ {full_url} Êú™ÊâæÂà∞ÊúÄÁªà‰∏ãËΩΩÈìæÊé•„ÄÇ")
            return
        file_url = final_link_tag['href']
        if "pan.baidu.com" in file_url:
            page_text = soup.get_text()
            match = re.search(r'ÊèêÂèñÁ†Å\s*[:Ôºö]\s*([a-zA-Z0-9]{4})', page_text)
            pass_code = match.group(1) if match else "Êú™ÊâæÂà∞"
            print(f"  - ÊèêÁ§∫ÔºöÊ£ÄÊµãÂà∞ÁôæÂ∫¶ÁΩëÁõòËµÑÊ∫êÔºåËØ∑ÊâãÂä®‰∏ãËΩΩ„ÄÇ")
            print(f"    - Ê†áÈ¢ò: {title}\n    - Âú∞ÂùÄ: {file_url}\n    - ÊèêÂèñÁ†Å: {pass_code}")
            return
        print(f"  ‚óè Ê≠£Âú®‰∏ãËΩΩÔºö{title}")
        file_response = requests.get(file_url, headers=HEADERS, stream=True)
        file_response.raise_for_status()
        file_name = f"{title}.zip"
        file_name = re.sub(r'[\\/*?:"<>|]', "", file_name)
        save_path = os.path.join(save_folder, file_name)
        with open(save_path, 'wb') as f:
            for chunk in file_response.iter_content(chunk_size=8192):
                f.write(chunk)
        print(f"    ‚úì ‰∏ãËΩΩÂÆåÊàêÔºåÂ∑≤‰øùÂ≠òËá≥: {save_path}")
    except Exception as e:
        print(f"  - ÈîôËØØÔºö‰∏ãËΩΩÊñá‰ª∂Êó∂Â§±Ë¥• - {e}")
