std::string rot13Cipher(const std::string& text) {
    std::string result = "";
    for (char c : text) {
        if (isalpha(c)) {
            char base = isupper(c) ? 'A' : 'a';
            // Apply ROT13 formula: (P + 13) % 26
            result += static_cast<char>(base + (c - base + 13) % 26);
        } else {
            result += c;
        }
    }
    return result;
