std::string atbashCipher(const std::string& text) {
    std::string result = "";
    for (char c : text) {
        if (isalpha(c)) {
            char base = isupper(c) ? 'A' : 'a';
            // Apply Atbash formula: C = (25 - P) or P = (25 - C)
            result += static_cast<char>(base + (25 - (c - base)));
        } else {
            result += c;
        }
    }
    return result;
}
